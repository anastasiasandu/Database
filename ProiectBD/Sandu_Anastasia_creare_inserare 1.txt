--Crearea tabelelor


CREATE TABLE Tara
(

ID INT IDENTITY(1,1) PRIMARY KEY,
Denumire VARCHAR(MAX)

)


CREATE TABLE Limba
(

ID INT IDENTITY(1,1) PRIMARY KEY,
Denumire VARCHAR(MAX)

)



CREATE TABLE Actor
(

ID INT IDENTITY(1,1) PRIMARY KEY,
Nume VARCHAR(MAX),
Varsta INT,
Sex CHAR(1),
TaraOrigineID INT REFERENCES Tara(ID)

)


CREATE TABLE Studio
(

ID INT IDENTITY(1,1) PRIMARY KEY,
Denumire VARCHAR(MAX),
SediuID INT REFERENCES Tara(ID)

)



CREATE TABLE Regizor
(

ID INT IDENTITY(1,1) PRIMARY KEY,
Nume VARCHAR(MAX),
Varsta INT,
Sex CHAR(1),
TaraOrigineID INT REFERENCES Tara(ID)

)


CREATE TABLE Scenarist
(

ID INT IDENTITY(1,1) PRIMARY KEY,
Nume VARCHAR(MAX),
Varsta INT,
Sex CHAR(1),
TaraOrigineID INT REFERENCES Tara(ID)

)


CREATE TABLE Categorie
(

ID INT IDENTITY(1,1) PRIMARY KEY,
Denumire VARCHAR(MAX)

)



CREATE TABLE Premiu
(

ID INT IDENTITY(1,1) PRIMARY KEY,
Denumire VARCHAR(MAX)

)


CREATE TABLE Film
(

ID INT IDENTITY(1,1) PRIMARY KEY,
Denumire VARCHAR(MAX) NOT NULL,
AnAparitie DATE NOT NULL,
IMDBRating DECIMAL(17,5),
Durata INT,
TaraOrigineID INT REFERENCES Tara(ID),
LimbaID INT REFERENCES Limba(ID)


)



CREATE TABLE FilmActorMapping
(

ID INT IDENTITY(1,1),
FilmID INT REFERENCES Film(ID),
ActorID INT REFERENCES Actor(ID)


Constraint PK_Film_Actor Primary Key (FilmID, ActorID)

)


CREATE TABLE FilmRegizorMapping
(

ID INT IDENTITY(1,1),
FilmID INT REFERENCES Film(ID),
RegizorID INT REFERENCES Regizor(ID)


CONSTRAINT PK_Film_Regizor PRIMARY KEY (FilmID, RegizorID)

)

CREATE TABLE FilmScenaristMapping
(

ID INT IDENTITY(1,1),
FilmID INT REFERENCES Film(ID),
ScenaristID INT REFERENCES Scenarist(ID)


CONSTRAINT PK_Film_Scenarist PRIMARY KEY (FilmID, ScenaristID)

)

CREATE TABLE FilmPremiuMapping
(

ID INT IDENTITY(1,1),
FilmID INT REFERENCES Film(ID),
PremiuID INT REFERENCES Premiu(ID)


CONSTRAINT PK_Film_Premiu PRIMARY KEY (FilmID, PremiuID)
)

CREATE TABLE ActorPremiuMapping
(

ID INT IDENTITY(1,1),
ActorID INT REFERENCES Actor(ID),
PremiuID INT REFERENCES Premiu(ID)


CONSTRAINT PK_Actor_Premiu PRIMARY KEY (ActorID, PremiuID)

)

CREATE TABLE RegizorPremiuMapping
(

ID INT IDENTITY(1,1),
RegizorID INT REFERENCES Regizor(ID),
PremiuID INT REFERENCES Premiu(ID)


CONSTRAINT PK_Regizor_Premiu PRIMARY KEY (RegizorID, PremiuID)

)

CREATE TABLE ScenaristPremiuMapping
(

ID INT IDENTITY(1,1),
ScenaristID INT REFERENCES Scenarist(ID),
PremiuID INT REFERENCES Premiu(ID)


CONSTRAINT PK_Scenarist_Premiu PRIMARY KEY (ScenaristID, PremiuID)

)
